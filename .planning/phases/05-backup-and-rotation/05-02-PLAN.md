```dart
/// Backup state constants for MPC key lifecycle tracking.
/// snake_case values match Rust Keygen/RecoveryCompletedPayload.backup_state.
/// Per D-04: Dart/host is responsible for state transitions and persistence (MPC-06).
abstract class BackupState {
  /// Keygen or rotation completed; backup envelope not yet created.
  static const String pending = 'pending';

  /// Host has stored a valid backup envelope.
  static const String backedUp = 'backed_up';

  /// Recovery (rotation) completed; previous backup envelope is invalid.
  /// Host must re-derive and store a new envelope with the new localEncryptedShare.
  static const String needsRotation = 'needs_rotation';
}
```

**жӯҘйӘӨ 2 вҖ” жӣҙж–° MpcClient.recover() ж”ҜжҢҒеҸҜйҖү backupпјҲD-03 и·Ҝеҫ„Aпјүпјҡ**

дҝ®ж”№ж–№жі•зӯҫеҗҚпјҢж–°еўһеҸҜйҖүеҸӮж•° `currentRotationVersion`пјҲдј з»ҷ Rustпјүе’ҢеҸҜйҖүеҸӮж•° `userBackupSecret`пјҲrotation еҗҺз”ҹжҲҗж–° envelopeпјүпјҡ

```dart
/// Execute full recovery protocol.
///
/// Decrypts the backup share, then drives recovery/rotation rounds.
/// Returns [RecoveryResult] with new localEncryptedShare and incremented rotationVersion.
///
/// Per D-03: If [userBackupSecret] is provided, a new backup envelope is derived
/// after rotation and returned in [RecoveryResult.encryptedBackupShare].
/// Host should update backupState to [BackupState.backedUp] after storing the envelope.
///
/// [currentRotationVersion]: current rotationVersion from host metadata (required for
/// correct version incrementing in Rust вҖ” fixes hardcoded bug).
Future<RecoveryResult> recover({
  required String mpcKeyId,
  required String encryptedBackupShare,
  required String userBackupSecret,
  required int currentRotationVersion,   // вҶҗ NEW: from host metadata
  String? newUserBackupSecret,           // вҶҗ NEW: if provided, derive new envelope after rotation
}) async {
  // Step 1: decrypt existing backup to get MasterKey2 JSON
  final backupShare = await _engine.decryptBackupShare(
    encryptedBackupShare,
    userBackupSecret,
  );

  // Step 2: initiate recovery with server
  final initResponse = await _sendToServer(
    '/recovery/start',
    jsonEncode({'mpcKeyId': mpcKeyId}),
  );
  final initData = _parseServerResponse(initResponse);
  final sessionId = initData['sessionId'] as String;

  // Step 3: Rust recover_start (now includes currentRotationVersion)
  final round1 = await _engine.recoverStart(
    sessionId,
    backupShare,
    jsonEncode(initData['serverPayload']),
    currentRotationVersion,    // вҶҗ pass to Rust for correct version tracking
  );
  _checkProtocolError(round1);

  // Step 4: drive rounds until completion
  var currentResult = round1;
  while (currentResult.isContinue) {
    final serverResponse = await _sendToServer(
      '/recovery/continue',
      jsonEncode({
        'sessionId': sessionId,
        'round': currentResult.round,
        'clientPayload': currentResult.clientPayload,
      }),
    );
    final serverData = _parseServerResponse(serverResponse);

    if (serverData['status'] == 'completed') {
      return RecoveryResult.fromJson(serverData);
    }

    currentResult = await _engine.recoverContinue(
      sessionId,
      jsonEncode(serverData['serverPayload']),
    );
    _checkProtocolError(currentResult);
  }

  if (currentResult.isCompleted && currentResult.clientPayload != null) {
    final payload =
        jsonDecode(currentResult.clientPayload!) as Map<String, dynamic>;
    final baseResult = RecoveryResult.fromJson(_snakeToCamelKeys(payload));

    // Step 5 (D-03 и·Ҝеҫ„A): optionally derive new backup envelope after rotation
    String? newEnvelopeJson;
    if (newUserBackupSecret != null) {
      final envelope = await _engine.deriveBackupEnvelope(
        baseResult.localEncryptedShare,
        newUserBackupSecret,
        DateTime.now().toIso8601String(),
      );
      newEnvelopeJson = jsonEncode(envelope.toJson());
    }

    return RecoveryResult(
      mpcKeyId: baseResult.mpcKeyId,
      address: baseResult.address,
      publicKey: baseResult.publicKey,
      rotationVersion: baseResult.rotationVersion,
      localEncryptedShare: baseResult.localEncryptedShare,
      encryptedBackupShare: newEnvelopeJson,  // null if newUserBackupSecret not provided
    );
  }

  throw MpcProtocolException(
      'Unexpected recovery state: ${currentResult.status}');
}
```
