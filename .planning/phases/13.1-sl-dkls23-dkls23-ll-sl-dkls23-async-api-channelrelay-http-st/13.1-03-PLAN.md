mod test_dkg;

use ceres_mpc::api::mpc_engine::{decrypt_backup_share, derive_backup_envelope, export_private_key};
use ceres_mpc::api::types::{BackupEnvelope, DecryptBackupResult, ExportResult};
use base64::engine::general_purpose::STANDARD as BASE64_STANDARD;
use base64::Engine;

/// AUX-01: Backup roundtrip with sl-dkls23 Keyshare bytes format
#[tokio::test]
async fn test_backup_roundtrip() {
    let (ks0, _ks1) = test_dkg::run_dkg_two_party().await;
    
    // Serialize keyshare to Base64 (new format)
    let keyshare_b64 = BASE64_STANDARD.encode(ks0.as_slice());
    
    // Encrypt
    let envelope_json = derive_backup_envelope(
        keyshare_b64.clone(),
        "test-backup-secret".to_string(),
        "2026-01-01T00:00:00Z".to_string(),
    ).expect("derive_backup_envelope must succeed");
    
    // Decrypt
    let result_json = decrypt_backup_share(
        envelope_json,
        "test-backup-secret".to_string(),
    ).expect("decrypt_backup_share must succeed");
    
    let result: DecryptBackupResult = serde_json::from_str(&result_json).unwrap();
    
    // Verify roundtrip: decode Base64 → Keyshare::from_bytes → same public key
    let restored_bytes = BASE64_STANDARD.decode(&result.device_backup_share).unwrap();
    let restored_ks = sl_dkls23::keygen::Keyshare::from_bytes(&restored_bytes)
        .expect("from_bytes must succeed");
    assert_eq!(restored_ks.public_key(), ks0.public_key());
}

#[test]
fn test_backup_wrong_secret_fails() {
    // This test doesn't need async — just test decrypt with wrong key
    let envelope_json = derive_backup_envelope(
        "some-data".to_string(),
        "correct-secret".to_string(),
        "2026-01-01T00:00:00Z".to_string(),
    ).unwrap();
    
    let result = decrypt_backup_share(envelope_json, "wrong-secret".to_string());
    assert!(result.is_err());
}

/// AUX-02: Key export using combine_shares
#[tokio::test]
async fn test_export_private_key() {
    let (ks0, ks1) = test_dkg::run_dkg_two_party().await;

    let share0_b64 = BASE64_STANDARD.encode(ks0.as_slice());
    let share1_b64 = BASE64_STANDARD.encode(ks1.as_slice());
    
    let result_json = export_private_key(share0_b64, share1_b64)
        .expect("export_private_key must succeed");
    
    let result: ExportResult = serde_json::from_str(&result_json).unwrap();
    assert!(result.exported);
    assert!(result.address.starts_with("0x"));
    assert_eq!(result.address.len(), 42);
    // private_key should be 64 hex chars (32 bytes)
    assert_eq!(result.private_key.len(), 64);
}
