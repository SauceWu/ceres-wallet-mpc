**Helper function `run_dkg_two_party() -> (Keyshare, Keyshare)`:**
Extract the full DKG protocol into a reusable helper so Phase 10/11 tests can call it to get Keyshares for DSG/Rotation testing.

```rust
use dkls23_ll::dkg::{State, Party, Keyshare};
use rand::thread_rng;

/// Run full 4-round DKG between two in-process parties.
/// Returns (party0_keyshare, party1_keyshare).
/// Reusable by DSG and Rotation tests.
fn run_dkg_two_party() -> (Keyshare, Keyshare) {
    let mut rng = thread_rng();
    let ranks = vec![0u8, 0u8];

    let mut p0 = State::new(Party { ranks: ranks.clone(), party_id: 0, t: 2 }, &mut rng, None);
    let mut p1 = State::new(Party { ranks: ranks.clone(), party_id: 1, t: 2 }, &mut rng, None);

    // Round 1
    let msg1_0 = p0.generate_msg1();
    let msg1_1 = p1.generate_msg1();

    // Round 2 (handle_msg1 — pass OTHER party's msg1 only)
    let msgs2_from_0 = p0.handle_msg1(&mut rng, vec![msg1_1.clone()]).unwrap();
    let msgs2_from_1 = p1.handle_msg1(&mut rng, vec![msg1_0.clone()]).unwrap();

    // Round 3: handle_msg2 (pass msg2 addressed to self)
    let msgs3_from_0 = p0.handle_msg2(&mut rng, vec![msgs2_from_1[0].clone()]).unwrap();
    let msgs3_from_1 = p1.handle_msg2(&mut rng, vec![msgs2_from_0[0].clone()]).unwrap();

    // Round 3a: calculate_commitment_2 (AFTER handle_msg2)
    let c2_0 = p0.calculate_commitment_2();
    let c2_1 = p1.calculate_commitment_2();
    let c2_list = vec![c2_0, c2_1]; // index == party_id

    // Round 4: handle_msg3
    let msg4_0 = p0.handle_msg3(&mut rng, vec![msgs3_from_1[0].clone()], &c2_list).unwrap();
    let msg4_1 = p1.handle_msg3(&mut rng, vec![msgs3_from_0[0].clone()], &c2_list).unwrap();

    // Complete: handle_msg4
    let share0 = p0.handle_msg4(vec![msg4_1.clone()]).unwrap();
    let share1 = p1.handle_msg4(vec![msg4_0.clone()]).unwrap();

    (share0, share1)
}
```
