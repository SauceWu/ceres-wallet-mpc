    import 'dart:typed_data';

    import 'package:flutter_mpc_wallet/src/rust/api/mpc_engine.dart' as rust_engine;
    import 'package:flutter_mpc_wallet/src/rust/api/types.dart';
    import 'package:flutter_mpc_wallet/src/rust/frb_generated.dart';

    /// Rust FFI wrapper 层 — 封装 FRB 生成的函数调用。
    /// 此类为 SDK 内部使用，不直接暴露给宿主（per D-06）。
    class MpcEngine {
      bool _initialized = false;

      /// 确保 FRB 原生库已初始化（Pitfall 6 防护）。
      Future<void> ensureInitialized() async {
        if (!_initialized) {
          await MpcBridge.init();
          _initialized = true;
        }
      }

      Future<MpcRoundResult> keygenStart(Uint8List serverPayload) async {
        await ensureInitialized();
        return await rust_engine.keygenStart(serverPayload: serverPayload);
      }

      Future<MpcRoundResult> keygenContinue(String sessionId, int round, Uint8List serverPayload) async {
        await ensureInitialized();
        return await rust_engine.keygenContinue(
          request: MpcSessionRequest(sessionId: sessionId, round: round, serverPayload: serverPayload),
        );
      }

      Future<MpcRoundResult> recoverStart(Uint8List serverPayload) async {
        await ensureInitialized();
        return await rust_engine.recoverStart(serverPayload: serverPayload);
      }

      Future<MpcRoundResult> recoverContinue(String sessionId, int round, Uint8List serverPayload) async {
        await ensureInitialized();
        return await rust_engine.recoverContinue(
          request: MpcSessionRequest(sessionId: sessionId, round: round, serverPayload: serverPayload),
        );
      }

      Future<MpcRoundResult> signStart(Uint8List serverPayload) async {
        await ensureInitialized();
        return await rust_engine.signStart(serverPayload: serverPayload);
      }

      Future<MpcRoundResult> signContinue(String sessionId, int round, Uint8List serverPayload) async {

    **Step 3: 创建 lib/src/client/mpc_client.dart（per D-05, D-06）**

    ```dart
    import 'dart:typed_data';

    import 'package:flutter_mpc_wallet/src/bridge/mpc_engine.dart';
    import 'package:flutter_mpc_wallet/src/rust/api/types.dart';
    import 'package:flutter_mpc_wallet/src/transport/mpc_transport.dart';

    /// MPC 钱包高级 API — 宿主直接使用的入口类（per D-05）。
    /// 注入 MpcTransport 实现网络请求，内部通过 MpcEngine 驱动 Rust 侧计算。
    /// 宿主无需了解 round 数量或协议时序（per D-02）。
    class MpcClient {
      final MpcTransport _transport;
      final MpcEngine _engine;

      MpcClient({
        required MpcTransport transport,
        MpcEngine? engine,
      })  : _transport = transport,
            _engine = engine ?? MpcEngine();

      /// 执行完整 keygen 协议。
      /// 返回值包含 mpcKeyId、address、publicKey 等（per architecture doc 0.7）。
      Future<KeygenCompleteResult> keygen() async {
        // Step 1: 向服务端发起 keygen start
        final startPayload = await _transport.send('/keygen/start', Uint8List(0));

        // Step 2: 将服务端 payload 交给 Rust engine 处理
        var round = await _engine.keygenStart(startPayload);

        // Step 3: 循环直到协议完成
        while (!round.isComplete) {
          final serverResp = await _transport.send('/keygen/continue', round.clientPayload);
          round = await _engine.keygenContinue(round.sessionId, round.round, serverResp);
        }

        // Phase 1 stub: 返回占位结果。Phase 3 将解析真实 keygen 完成数据。
        return KeygenCompleteResult(
          mpcKeyId: 'stub_key_id',
          address: '0x0000000000000000000000000000000000000000',
          publicKey: '0x00',
          curve: 'secp256k1',
          threshold: 2,
          keyRef: 'stub_key_ref',
          rotationVersion: 1,
          localEncryptedShare: round.clientPayload,
        );
      }

      /// 执行完整 recovery 协议。
      /// [mpcKeyId] — 要恢复的密钥 ID（per MPC-08: recovery 输入只收 mpcKeyId）
      Future<RecoverCompleteResult> recover(String mpcKeyId) async {
        final startPayload = await _transport.send(
          '/recovery/start',
          Uint8List.fromList(mpcKeyId.codeUnits),
        );

        var round = await _engine.recoverStart(startPayload);

        while (!round.isComplete) {
          final serverResp = await _transport.send('/recovery/continue', round.clientPayload);
          round = await _engine.recoverContinue(round.sessionId, round.round, serverResp);
        }

        return RecoverCompleteResult(
          mpcKeyId: mpcKeyId,
          address: '0x0000000000000000000000000000000000000000',
          publicKey: '0x00',
          rotationVersion: 1,
          localEncryptedShare: round.clientPayload,
        );
      }

      /// 执行完整 sign 协议。
      /// [mpcKeyId] — 使用的密钥 ID
      /// [signPayload] — 待签名数据
      Future<SignCompleteResult> sign(String mpcKeyId, Uint8List signPayload) async {
        final startPayload = await _transport.send('/sign/start', signPayload);

        var round = await _engine.signStart(startPayload);

        while (!round.isComplete) {
          final serverResp = await _transport.send('/sign/continue', round.clientPayload);
          round = await _engine.signContinue(round.sessionId, round.round, serverResp);
        }

        return SignCompleteResult(
          signature: '0x00',
          signedTx: null,
          txHash: null,
        );
      }
    }
    ```

    import 'package:flutter_test/flutter_test.dart';
    import 'package:mocktail/mocktail.dart';
    import 'package:flutter_mpc_wallet/src/bridge/mpc_engine.dart';
    import 'package:flutter_mpc_wallet/src/client/mpc_client.dart';

    class MockMpcTransport extends Mock implements MpcTransport {}

    void main() {
      late MockMpcEngine mockEngine;
      late MockMpcTransport mockTransport;
      late MpcClient client;

      setUp(() {
        mockEngine = MockMpcEngine();
        mockTransport = MockMpcTransport();
        client = MpcClient(transport: mockTransport, engine: mockEngine);
      });

      setUpAll(() {
        registerFallbackValue(Uint8List(0));
      });

      group('keygen', () {
        test('drives round-trip loop until engine returns isComplete=true', () async {
          // Mock transport responses
          when(() => mockTransport.send('/keygen/start', any()))
              .thenAnswer((_) async => Uint8List.fromList([1, 2, 3]));
          when(() => mockTransport.send('/keygen/continue', any()))
              .thenAnswer((_) async => Uint8List.fromList([4, 5, 6]));

          // Round 1: not complete → triggers continue
          when(() => mockEngine.keygenStart(any()))
              .thenAnswer((_) async => MpcRoundResult(
                    sessionId: 'sess_1',
                    round: 1,
                    clientPayload: Uint8List(32),
                    isComplete: false,
                  ));

          // Round 2: complete → exits loop
          when(() => mockEngine.keygenContinue(any(), any(), any()))
              .thenAnswer((_) async => MpcRoundResult(
                    sessionId: 'sess_1',
                    round: 2,
                    clientPayload: Uint8List(32),
                    isComplete: true,
                  ));

          final result = await client.keygen();

          expect(result.mpcKeyId, isNotEmpty);
          expect(result.address, startsWith('0x'));
          expect(result.curve, 'secp256k1');
          verify(() => mockTransport.send('/keygen/start', any())).called(1);
          verify(() => mockTransport.send('/keygen/continue', any())).called(1);
          verify(() => mockEngine.keygenStart(any())).called(1);
          verify(() => mockEngine.keygenContinue(any(), any(), any())).called(1);
        });
      });

      group('recover', () {
        test('sends mpcKeyId and drives round-trip loop', () async {
          when(() => mockTransport.send('/recovery/start', any()))

          when(() => mockTransport.send('/recovery/continue', any()))
              .thenAnswer((_) async => Uint8List.fromList([30, 40]));

          when(() => mockEngine.recoverStart(any()))

                    round: 1,
                    clientPayload: Uint8List(32),
                    isComplete: false,

                    isComplete: true,

          expect(result.mpcKeyId, 'key_123');
          expect(result.address, startsWith('0x'));
          verify(() => mockTransport.send('/recovery/start', any())).called(1);
          verify(() => mockEngine.recoverStart(any())).called(1);
        });

        test('sends sign payload and drives round-trip loop', () async {
