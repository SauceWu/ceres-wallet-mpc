```rust
use multi_party_ecdsa::protocols::two_party_ecdsa::lindell_2017::{party_one, party_two};
use kms_secp256k1::ecdsa::two_party::MasterKey2;

/// Sign session state persisted between rounds.
/// Holds Party2 ephemeral key state from round 1, plus Party1's ephemeral
/// first message (received from server in round 1) needed for sign_second_message.
pub struct SignSession {
    pub master_key: MasterKey2,
    pub eph_ec_key_pair: party_two::EphEcKeyPair,
    pub eph_comm_witness: party_two::EphCommWitness,
    pub eph_party1_first_message: party_one::EphKeyGenFirstMsg,

}

pub static SIGN_SESSIONS: Lazy<Mutex<HashMap<String, SignSession>>> =
    Lazy::new(|| Mutex::new(HashMap::new()));

pub fn remove_sign_session(session_id: &str) -> Option<SignSession> {
    SIGN_SESSIONS.lock().unwrap().remove(session_id)
}
```

3. **替换 sign_start stub：**
```rust
pub fn sign_start(
    session_id: String,
    share: String,
    server_payload: String,
) -> Result<String, String> {
    let server: SignRound1ServerPayload = serde_json::from_str(&server_payload)
        .map_err(|e| format!("invalid server_payload JSON: {e}"))?;

    // 防御性校验：messageHash 必须是 32 bytes hex (无 0x 前缀)
    let msg_bytes = hex::decode(&server.message_hash)
        .map_err(|e| format!("invalid message_hash hex: {e}"))?;
    if msg_bytes.len() != 32 {
        return Err(format!(
            "message_hash must be 32 bytes, got {}",
            msg_bytes.len()
        ));
    }

    // 反序列化 share → MasterKey2
    let master_key: MasterKey2 = serde_json::from_str(&share)
        .map_err(|e| format!("invalid share JSON: {e}"))?;

    // Party2 round 1：生成 ephemeral key pair
    let (eph_key_gen_first_message_party_two, eph_comm_witness, eph_ec_key_pair) =
        MasterKey2::sign_first_message();

    // 存入 session（含 Party1 ephemeral first message + messageHash）
    SIGN_SESSIONS.lock().unwrap().insert(
        session_id.clone(),
        SignSession {
            master_key,
            eph_ec_key_pair,
            eph_comm_witness,
            eph_party1_first_message: server.eph_key_gen_first_message_party_one,
            message_hash: server.message_hash,
        },
    );

    let client_payload = SignRound1ClientPayload {
        eph_key_gen_first_message_party_two,
    };

    let result = MpcRoundResult {
        status: "continue".to_string(),
        round: 1,
        client_payload: Some(
            serde_json::to_string(&client_payload).map_err(|e| e.to_string())?,
        ),
        error_message: None,
    };
    serde_json::to_string(&result).map_err(|e| e.to_string())
}
```

4. **替换 sign_continue stub：**
```rust
pub fn sign_continue(session_id: String, server_payload: String) -> Result<String, String> {
    // server_payload 在 round2 可为 {} — sign_second_message 不依赖额外服务端数据
    let _ = serde_json::from_str::<serde_json::Value>(&server_payload)
        .map_err(|e| format!("invalid server_payload JSON: {e}"))?;

    // 取出 session（remove 后 move 所有字段，EphCommWitness non-Copy by value）
    let session = remove_sign_session(&session_id)
        .ok_or_else(|| format!("session not found: {session_id}"))?;

    // 重建 message BigInt
    let msg_bytes = hex::decode(&session.message_hash)
        .map_err(|e| format!("invalid stored message_hash: {e}"))?;
    let message = BigInt::from_bytes(&msg_bytes);

    // Party2 round 2：计算 partial sig（注意 eph_comm_witness by value）
    let sign_party_two_second_message = session.master_key.sign_second_message(
        &session.eph_ec_key_pair,
        session.eph_comm_witness,         // moved, not cloned
        &session.eph_party1_first_message,
        &message,
    );

    // sign_party_two_second_message 是 SignMessage { partial_sig, second_message }
    // 序列化后作为 client_payload 发给服务端（服务端用来调用 Party1 sign_second_message）
    // 注意：Party2 侧不持有最终 SignatureRecid；由服务端完成并以 completed 返回
    // 但为保持 SDK 自测完整性，此处以 completed 直接返回 SignMessage JSON
    // 实际部署中服务端会解析此 SignMessage 并返回 r/s/recid
    let client_payload_str = serde_json::to_string(&sign_party_two_second_message)
        .map_err(|e| e.to_string())?;

    let result = MpcRoundResult {
        status: "completed".to_string(),
        round: 2,
        client_payload: Some(client_payload_str),
        error_message: None,
    };
    serde_json::to_string(&result).map_err(|e| e.to_string())
}
```

**重要说明关于 sign_continue 的 completed payload：**
sign_continue 返回的 client_payload 是 Party2 的 SignMessage（partial_sig + second_message），而非最终的 SignatureRecid。最终 SignatureRecid 由服务端（Party1）通过 sign_second_message 计算并以 `status: completed` + `{ r, s, recid }` 格式返回给 Dart 层。

MpcClient.sign() 在收到服务端 `status: completed` 时直接解析 `{ r, s, recid }`，而不是从 Rust sign_continue 的 client_payload 中解析。因此 `sign_continue` 的 completed 语义是"Party2 协议工作完成，输出已发出"。

```rust
fn test_sign_full_protocol() {
    // 1. Keygen 产生 MasterKey1 + MasterKey2
```
